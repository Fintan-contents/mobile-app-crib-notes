Button

各ネイティブプラットフォームの標準スタイルを使用したボタンを表示する場合、このコンポーネントを使用します。 色以外のスタイルがカスタマイズ出来ないため、デザインがアプリに適していない場合、代わりにTouchablesやPressableを使用して独自のコンポーネントを作成します。

  • 画面イメージ
  • ソースコード

/src/App.tsx
import React, {useState} from 'react';
import {Button, StyleSheet, View} from 'react-native';

export const App = () => {
  const [count, setCount] = useState(0);

  return (
    <View style={styles.container}>
      <Button
        title={`count is ${count}`}
        onPress={() => {
          setCount(count + 1);
        }}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
  },
});
